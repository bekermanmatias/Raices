def newton_raphson(f, df, x0, tolerancia=1e-6, max_iter=100):
    """
    M√©todo de Newton-Raphson para encontrar ra√≠ces
    f: funci√≥n
    df: derivada de la funci√≥n
    x0: valor inicial
    """
    x = x0
    for i in range(max_iter):
        fx = f(x)
        if abs(fx) < tolerancia:
            return x, i+1
        dfx = df(x)
        if dfx == 0:
            return None, i+1
        x = x - fx/dfx
    return x, max_iter

# Definir funci√≥n y su derivada
def f(x):
    return x**2 - 5*x + 6

def df(x):
    return 2*x - 5

# Encontrar ra√≠ces

raiz2, iter2 = newton_raphson(f, df, 4)

print(f"Ra√≠z 2: {raiz2:.6f} (en {iter2} iteraciones)")
